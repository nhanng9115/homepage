<script>
(function(){
  // Load optional overrides
  let BIB_DB = {};
  try { BIB_DB = JSON.parse(document.getElementById('bibtex-db')?.textContent || '{}'); } catch(e){}

  // Helpers
  function quotedTitle(li){
    const m = li.innerHTML.match(/"([^"]{3,})"/);
    return m ? m[1].trim() : null;
  }
  function firstLink(li){
    const a = li.querySelector('a[href]');
    return a ? a.getAttribute('href') : null;
  }
  function text(li){ return li.textContent.replace(/\s+/g,' ').trim(); }

  // Create a decent fallback BibTeX (now includes url if present)
  function fallbackBib(li, title){
    const t = text(li);
    const before = title ? t.split(' "'+title+'"')[0] : t;
    const authors = before.replace(/,\s*$/,'');
    const year = (t.match(/,\s*(20\d{2})\./) || [,''])[1];
    const venue = (t.match(/,\s*([A-Za-z].*?)\s*,\s*20\d{2}/) || [,''])[1].replace(/\(\*\*.*?\*\*\)/g,'').trim();
    const url = firstLink(li);
    const isJournal = /Transactions|Journal/i.test(venue);
    const key = (authors.split(',')[0]||'key').trim().split(' ').pop() + (year||'') + 'Auto';

    const base = isJournal ? 
`@article{${key},
  author = {${authors}},
  title  = {${title}},
  journal= {${venue}},
  year   = {${year}}` :
`@inproceedings{${key},
  author   = {${authors}},
  title    = {${title}},
  booktitle= {${venue}},
  year     = {${year}}`;

    const tail = url ? `,\n  url    = {${url}}\n}` : `\n}`;
    return base + tail;
  }

  function buildPanel(bibtex){
    const box = document.createElement('div');
    box.className = 'bibtex-box';
    const copy = document.createElement('button');
    copy.className = 'bibtex-copy';
    copy.type = 'button';
    copy.textContent = 'Copy';
    copy.addEventListener('click', ()=>{
      navigator.clipboard.writeText(bibtex).then(()=>{
        const t = copy.textContent; copy.textContent='Copied!'; setTimeout(()=>copy.textContent=t, 1200);
      });
    });
    const pre = document.createElement('pre');
    const code = document.createElement('code');
    code.textContent = bibtex;
    pre.appendChild(code);
    box.appendChild(copy);
    box.appendChild(pre);
    return box;
  }

  // Add a single button to each <li>
  document.querySelectorAll('ol > li').forEach(li=>{
    const btn = document.createElement('button');
    btn.className = 'bibtex-btn';
    btn.type = 'button';
    btn.textContent = 'BibTex';
    btn.addEventListener('click', ()=>{
      document.querySelectorAll('.bibtex-box').forEach(b=>b.remove());
      const title = quotedTitle(li);
      let bib = (title && BIB_DB[title]) ? BIB_DB[title] :
                title ? fallbackBib(li, title) :
                '@misc{citation,\n  note={Unable to auto-detect title for this entry}\n}';
      const panel = buildPanel(bib);
      btn.insertAdjacentElement('afterend', panel);
      panel.scrollIntoView({behavior:'smooth', block:'nearest'});
    });
    li.appendChild(document.createElement('br'));
    li.appendChild(btn);
  });
})();
